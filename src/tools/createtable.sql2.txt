■テーブル作成
CREATE TABLE team(id int not null primary key, parentID int references team(id), name varchar(40));
CREATE TABLE member(id int not null primary key, groupID int references team(id), name varchar(40));
CREATE TABLE event(id int not null primary key, name varchar(40));
CREATE TABLE eventDate(id int not null primary key, eventID int references event(id) not null, date date not null, unique(id, date));
CREATE TABLE eventAttendance(userID int references member(id) not null, eventDateID int references eventDate(id) not null, attendance varchar(10), primary key(userID, eventDateID));
CREATE TABLE eventComment(userID int references member(id) not null, eventID int references event(id) not null, comment varchar(40), primary key(userID, eventID));
